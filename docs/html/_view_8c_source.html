173 static void notifyObservers(View *self, Notification *notification) {
174  pthread_rwlock_rdlock(&observerMap_mutex);
175  ObserverMap *pointer = self->observerMap;
176 
177  while (pointer) {
178  if (strcmp(pointer->name, notification->name) == 0) {
179  // Get a reference to the observers list for this notification name
180  ObserverNode *cursor = pointer->observers;
181 
182  // Copy observers from reference array to working array,
183  // since the reference array may change during the notification loop
184  ObserverNode *observers = NULL;
185  ObserverNode **previous = &observers;
186  while (cursor) {
187  Observer *observer = NewObserver(cursor->observer->notify, cursor->observer->context);
188  *previous = NewObserverNode(observer);
189  previous = &(*previous)->next;
190  cursor = cursor->next;
191  }
192 
193  // Notify Observers from the working array
194  while (observers) {
195  Observer *observer = observers->observer;
196  observer->notifyObserver(observer, notification);
197  ObserverNode *next = observers->next;
198  DeleteObserverNode(observers);
199  observers = next;
200  }
201  break;
202  }
203  pointer = pointer->next;
204  }
205  pthread_rwlock_unlock(&observerMap_mutex);

208 static void removeObserver(View *self, const char *notificationName, void *notifyContext) {
209  pthread_rwlock_wrlock(&observerMap_mutex);
210  ObserverMap **observerMap = &self->observerMap;
211 
212  while (*observerMap) {
213  if (strcmp((*observerMap)->name, notificationName) == 0) {
214  // the observer list for the notification under inspection
215  ObserverNode **observers = &(*observerMap)->observers;
216 
217  // find the observer for the notifyContext
218  while (*observers) {
219  Observer *observer = (*observers)->observer;
220  if (observer->compareNotifyContext(observer, notifyContext)) {
221  // there can only be one Observer for a given notifyContext
222  // in any given Observer list, so remove it and break
223  ObserverNode *node = *observers;
224  *observers = (*observers)->next; // delete observer
225  DeleteObserverNode(node);
226 
227  // Also, when a Notification's Observer list length falls to
228  // zero, delete the notification key from the observer map
229  if (CountObservers(*observers) == 0) {
230  ObserverMap *map = *observerMap;
231  *observerMap = (*observerMap)->next;
232  DeleteObserverMap(map);
233  }
234  return;
235  }
236  observers = &(*observers)->next;
237  }
238  }
239  observerMap = &(*observerMap)->next;
240  }
241  pthread_rwlock_unlock(&observerMap_mutex);
