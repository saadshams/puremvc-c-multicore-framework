48static void notifyObservers(const struct IView *self, const struct INotification *notification) {
49 struct View *this = (struct View *) self;
50 struct IArray *copy = NULL;
51
52 mutex_lock_shared(&this->observerMapMutex);
53 if (this->observerMap->containsKey(this->observerMap, notification->getName(notification))) {
54 // Get a reference to the clone list for this notification name
55 const struct IArray *observers = (struct IArray *) this->observerMap->get(this->observerMap, notification->getName(notification));
56
57 // Copy observers from reference array to working array,
58 // since the reference array may change during the notification loop
59 copy = observers->clone(observers);
60 }
61 mutex_unlock(&this->observerMapMutex);
62
63 if (copy != NULL) {
64 copy->forEach(copy, notifyObserver, notification);
65 copy->clear(copy, NULL);
66 collection_array_free(&copy);
67 }
68}

75static void removeObserver(const struct IView *self, const char *notificationName, const void *notifyContext) {
76 struct View *this = (struct View *) self;
77
78 mutex_lock(&this->observerMapMutex);
79 struct IArray *observers = (struct IArray *) this->observerMap->get(this->observerMap, notificationName);
80 struct IObserver *observer = (struct IObserver *) observers->find(observers, compareNotifyContext, notifyContext);
81 if (observer != NULL) {
82 observers->removeItem(observers, observer);
83 puremvc_observer_free(&observer);
84 }
85 if (observers->size(observers) == 0) {
86 this->observerMap->removeItem(this->observerMap, notificationName);
87 }
88 mutex_unlock(&this->observerMapMutex);

91static void registerMediator(const struct IView *self, struct IMediator *mediator) {
92 struct View *this = (struct View *) self;
93
94 mutex_lock(&this->mediatorMapMutex);
95 if (this->mediatorMap->containsKey(this->mediatorMap, mediator->getName(mediator))) {
96 mutex_unlock(&this->mediatorMapMutex);
97 return;
98 }
99
100 mediator->notifier->initializeNotifier(mediator->notifier, this->multitonKey);
101 this->mediatorMap->put(this->mediatorMap, mediator->getName(mediator), mediator);
102 mutex_unlock(&this->mediatorMapMutex);
103
104 char **interests = mediator->listNotificationInterests(mediator);
105 for(char **cursor = interests; *cursor; cursor++) {
106 const char *error = NULL;
107 const struct IObserver *observer = puremvc_observer_new((const void (*)(const void *, struct INotification *)) mediator->handleNotification, mediator, &error);
108 self->registerObserver(self, *cursor, observer);
109 }
110 mediator->freeNotificationInterests(mediator, interests);
111 mediator->onRegister(mediator);
112}

130static struct IMediator *removeMediator(const struct IView *self, const char *mediatorName) {
131 struct View *this = (struct View *) self;
132
133 mutex_lock(&this->mediatorMapMutex);
134 struct IMediator *mediator = this->mediatorMap->removeItem(this->mediatorMap, mediatorName);
135 mutex_unlock(&this->mediatorMapMutex);
136
137 if (mediator != NULL) {
138 char **interests = mediator->listNotificationInterests(mediator);
139 for (char **cursor = interests; *cursor; cursor++) {
140 self->removeObserver(self, *cursor, mediator);
141 }
142 mediator->freeNotificationInterests(mediator, interests);
143 mediator->onRemove(mediator);
144 }

162static struct View *alloc(const char *key, const char **error) {
163 if (instanceMap->containsKey(instanceMap, key)) {
164 fprintf(stderr, "[PureMVC::View::alloc] View instance with key '%s' already exists!\n", key);
165 exit(EXIT_FAILURE);
166 }
167
168 struct View *view = malloc(sizeof(struct View));
169 if (view == NULL)
170 return *error = "[PureMVC::View::alloc] Error: Failed to allocate View", NULL;
171
172 memset(view, 0, sizeof(struct View));
173
174 view->multitonKey = strdup(key);
175 if (view->multitonKey == NULL)
176 return *error = "[PureMVC::View::alloc] Error: Failed to allocate View key (strdup).", free(view), NULL;
177
178 mutex_init(&view->mediatorMapMutex);
179 view->mediatorMap = collection_dictionary_new();
180 mutex_init(&view->observerMapMutex);
181 view->observerMap = collection_dictionary_new();
182 return view;
183}

210struct IView *puremvc_view_getInstance(const char *key, struct IView *(*factory)(const char *key, const char **error)) {
211 if (key == NULL || factory == NULL) return NULL;
212 mutex_once(&token, dispatchOnce);
213 mutex_lock(&mutex);
214
215 if (instanceMap == NULL) instanceMap = collection_dictionary_new();
216
217 struct IView *instance = (struct IView *) instanceMap->get(instanceMap, key);
218 if (instance == NULL) {
219 const char *error = NULL;
220 instance = factory(key, &error);
221 instanceMap->put(instanceMap, key, instance);
222 instance->initializeView(instance);
223 }
224
225 mutex_unlock(&mutex);
226 return instance;
227}
