41static void notifyObservers(const struct IView *self, const struct INotification *notification) {
42 struct View *this = (struct View *) self;
43 struct IArray *copy = NULL;
44
45 mutex_lock_shared(&this->observerMapMutex);
46 if (this->observerMap->containsKey(this->observerMap, notification->getName(notification))) {
47 // Get a reference to the clone list for this notification name
48 const struct IArray *observers = (struct IArray *) this->observerMap->get(this->observerMap, notification->getName(notification));
49
50 // Copy observers from reference array to working array,
51 // since the reference array may change during the notification loop
52 copy = observers->clone(observers);
53 }
54 mutex_unlock(&this->observerMapMutex);
55
56 if (copy != NULL) {
57 copy->forEach(copy, notifyObserver, notification);
58 copy->clear(copy, NULL);
59 collection_array_free(&copy);
60 }
61}

68static void removeObserver(const struct IView *self, const char *notificationName, const void *notifyContext) {
69 struct View *this = (struct View *) self;
70
71 mutex_lock(&this->observerMapMutex);
72 struct IArray *observers = (struct IArray *) this->observerMap->get(this->observerMap, notificationName);

84static void registerMediator(const struct IView *self, struct IMediator *mediator) {
85 struct View *this = (struct View *) self;
86
87 mutex_lock(&this->mediatorMapMutex);
88 if (this->mediatorMap->containsKey(this->mediatorMap, mediator->getName(mediator))) {
89 mutex_unlock(&this->mediatorMapMutex);
90 return;
91 }
92
93 mediator->notifier->initializeNotifier(mediator->notifier, this->multitonKey);
94 this->mediatorMap->put(this->mediatorMap, mediator->getName(mediator), mediator);
95 mutex_unlock(&this->mediatorMapMutex);
96
97 char **interests = mediator->listNotificationInterests(mediator);
98 for(char **cursor = interests; *cursor; cursor++) {
99 const struct IObserver *observer = puremvc_observer_new((const void (*)(const void *, struct INotification *)) mediator->handleNotification, mediator);
100 self->registerObserver(self, *cursor, observer);
101 }
102 mediator->freeNotificationInterests(mediator, interests);
103 mediator->onRegister(mediator);
104}
