167struct IFacade *puremvc_facade_getInstance(const char *key, struct IFacade *(*factory)(const char *, const char **error), const char **error) {
168 if (key == NULL || factory == NULL) return *error = "[PureMVC::Facade::getInstance] Error: key or factory must not be NULL.", NULL;
169 mutex_once(&token, dispatchOnce);
170 mutex_lock(&mutex);
171
172 if (instanceMap == NULL) {
173 instanceMap = collection_dictionary_new(error);
174 if (*error != NULL) return NULL;
175 }
176
177 struct IFacade *instance = (struct IFacade *) instanceMap->get(instanceMap, key);
178 if (instance == NULL) {
179 instance = factory(key, error);
180 if (*error != NULL) return mutex_unlock(&mutex), NULL;
181
182 instance->initializeFacade(instance, error);
183 if (*error != NULL) return puremvc_facade_free(&instance), mutex_unlock(&mutex), NULL;
184
185 instanceMap->put(instanceMap, key, instance, error);
186 if (*error != NULL) return puremvc_facade_free(&instance), mutex_unlock(&mutex), NULL;
187 }
188
189 mutex_unlock(&mutex);
190 return instance;
191}

193bool puremvc_facade_hasCore(const char *key) {

201void puremvc_facade_removeFacade(const char *key) {
202 if (key == NULL) return;
203 mutex_lock(&mutex);
204
205 puremvc_model_removeModel(key);
206 puremvc_view_removeView(key);
207 puremvc_controller_removeController(key);
208
209 struct IFacade *facade = instanceMap->removeItem(instanceMap, key);
210 if (facade != NULL) puremvc_facade_free(&facade);
211
212 mutex_unlock(&mutex);
213}
