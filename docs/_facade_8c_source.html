137static struct Facade *alloc(const char *key, const char **error) {
138 if (instanceMap->containsKey(instanceMap, key)) {
139 fprintf(stderr, "[PureMVC::Facade::alloc] Facade instance with key '%s' already exists!\n", key);
140 exit(EXIT_FAILURE);
141 }
142
143 struct Facade *facade = malloc(sizeof(struct Facade));
144 if (facade == NULL)
145 return *error = "[PureMVC::Facade::alloc] Error: Failed to allocate Facade", NULL;
146
147 memset(facade, 0, sizeof(*facade));
148
149 facade->multitonKey = strdup(key);
150 if (facade->multitonKey == NULL)
151 return *error = "[PureMVC::Facade::alloc] Error: Failed to allocate Facade key (strdup).", free(facade), NULL;
152
153 return facade;
154}

175struct IFacade *puremvc_facade_getInstance(const char *key, struct IFacade *(*factory)(const char *, const char **error)) {
176 if (key == NULL || factory == NULL) return NULL;
177 mutex_once(&token, dispatchOnce);
178 mutex_lock(&mutex);
179
180 if (instanceMap == NULL) instanceMap = collection_dictionary_new();
181
182 struct IFacade *instance = (struct IFacade *) instanceMap->get(instanceMap, key);
183 if (instance == NULL) {
184 const char *error = NULL;
185 instance = factory(key, &error);
186 instanceMap->put(instanceMap, key, instance);
187 instance->initializeFacade(instance);
188 }
189
190 mutex_unlock(&mutex);
191 return instance;
192}

194bool puremvc_facade_hasCore(const char *key) {
195 if (key == NULL) return false;
196 mutex_lock_shared(&mutex);
197 const bool result = instanceMap->containsKey(instanceMap, key);
198 mutex_unlock(&mutex);
199 return result;
200}

202void puremvc_facade_removeFacade(const char *key) {
203 if (key == NULL) return;
204 mutex_lock(&mutex);
205
206 puremvc_model_removeModel(key);
207 puremvc_view_removeView(key);
208 puremvc_controller_removeController(key);
209
210 struct IFacade *facade = instanceMap->removeItem(instanceMap, key);
211 if (facade != NULL) puremvc_facade_free(&facade);
212
213 mutex_unlock(&mutex);
214}
