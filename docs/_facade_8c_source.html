168struct IFacade *puremvc_facade_getInstance(const char *key, struct IFacade *(*factory)(const char *)) {
169 assert(key != NULL);
170 assert(factory != NULL);
171 mutex_once(&token, dispatchOnce);
172 mutex_lock(&mutex);
173
174 if (instanceMap == NULL) instanceMap = collection_dictionary_new();
175
176 struct IFacade *instance = (struct IFacade *) instanceMap->get(instanceMap, key);
177 if (instance == NULL) {
178 instance = factory(key);
179 instanceMap->put(instanceMap, key, instance);
180 instance->initializeFacade(instance);
181 }
182
183 mutex_unlock(&mutex);
184 return instance;
185}

187bool puremvc_facade_hasCore(const char *key) {

190 const bool result = instanceMap->containsKey(instanceMap, key);
191 mutex_unlock(&mutex);
192 return result;
