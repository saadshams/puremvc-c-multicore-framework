29static void executeCommand(const struct IController *self, struct INotification *notification, const char **error) {
30 struct Controller *this = (struct Controller *) self;
31
32 mutex_lock_shared(&this->commandMapMutex);
33 struct ICommand *(*factory)(const char **error) = (struct ICommand *(*)(const char **)) this->commandMap->get(this->commandMap, notification->getName(notification));
34 mutex_unlock(&this->commandMapMutex);
35
36 if (factory != NULL) {
37 struct ICommand *command = factory(error);

43 command->execute(command, notification, error);

45

47 }
48}

50static void registerCommand(const struct IController *self, const char *notificationName, struct ICommand *(*factory)(const char **error), const char **error) {
51 struct Controller *this = (struct Controller *) self;
52
53 mutex_lock(&this->commandMapMutex);
54 if (this->commandMap->containsKey(this->commandMap, notificationName) == false) {
55 const struct IObserver *observer = puremvc_observer_new((const void (*)(const void *, struct INotification *))executeCommand, this, error);
56 if (*error != NULL) return mutex_unlock(&this->commandMapMutex), (void)0;
57
58 this->view->registerObserver(this->view, notificationName, observer, error);
59 if (*error != NULL) return mutex_unlock(&this->commandMapMutex), (void)0;
60
61 this->commandMap->put(this->commandMap, notificationName, factory, error);
62 if (*error != NULL) return mutex_unlock(&this->commandMapMutex), (void)0;
63 } else {
64 this->commandMap->replace(this->commandMap, notificationName, factory);
65 }
66 mutex_unlock(&this->commandMapMutex);
67}

138struct IController *puremvc_controller_getInstance(const char *key, struct IController *(*factory)(const char *key, const char **error), const char **error) {
139 if (key == NULL || factory == NULL) return *error = "[PureMVC::Controller::getInstance] Error: key or factory must not be NULL.", NULL;
140 mutex_once(&token, dispatchOnce);
141 mutex_lock(&mutex);
142
143 if (instanceMap == NULL) {
144 instanceMap = collection_dictionary_new(error);
145 if (*error != NULL) return NULL;
146 }
147
148 struct IController *instance = (struct IController *) instanceMap->get(instanceMap, key);
149 if (instance == NULL) {
150 instance = factory(key, error);
151 if (*error != NULL) return mutex_unlock(&mutex), NULL;
152
153 instance->initializeController(instance, error);
154 if (*error != NULL) return puremvc_controller_free(&instance), mutex_unlock(&mutex), NULL;
155
156 instanceMap->put(instanceMap, key, instance, error);
157 if (*error != NULL) return puremvc_controller_free(&instance), mutex_unlock(&mutex), NULL;
158 }
159
160 mutex_unlock(&mutex);
161 return instance;
162}
