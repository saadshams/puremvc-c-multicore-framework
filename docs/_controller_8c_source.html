29static void executeCommand(const struct IController *self, struct INotification *notification) {
30 struct Controller *this = (struct Controller *) self;
31
32 mutex_lock_shared(&this->commandMapMutex);
33 struct ICommand *(*factory)() = (struct ICommand *(*)()) this->commandMap->get(this->commandMap, notification->getName(notification));
34 mutex_unlock(&this->commandMapMutex);

44static void registerCommand(const struct IController *self, const char *notificationName, struct ICommand *(*factory)()) {
45 struct Controller *this = (struct Controller *) self;
46
47 mutex_lock(&this->commandMapMutex);
48 if (this->commandMap->containsKey(this->commandMap, notificationName) == false) {
49 const char *error = NULL;
50 const struct IObserver *observer = puremvc_observer_new((const void (*)(const void *, struct INotification *))executeCommand, this, &error);
51 this->view->registerObserver(this->view, notificationName, observer);
52 this->commandMap->put(this->commandMap, notificationName, factory);
53 } else {
54 this->commandMap->replace(this->commandMap, notificationName, factory);
55 }
56 mutex_unlock(&this->commandMapMutex);

88static struct Controller *alloc(const char *key, const char **error) {

99
100 controller->multitonKey = strdup(key);
101 if (controller->multitonKey == NULL)
102 return *error = "[PureMVC::Facade::alloc] Error: Failed to allocate Controller key (strdup).", free(controller), NULL;

131struct IController *puremvc_controller_getInstance(const char *key, struct IController *(*factory)(const char *key, const char **error)) {
132 if (key == NULL || factory == NULL) return NULL;
133 mutex_once(&token, dispatchOnce);
134 mutex_lock(&mutex);
135
136 if (instanceMap == NULL) instanceMap = collection_dictionary_new();
137
138 struct IController *instance = (struct IController *) instanceMap->get(instanceMap, key);
139 if (instance == NULL) {
140 const char *error = NULL;
141 instance = factory(key, &error);
142 instanceMap->put(instanceMap, key, instance);
143 instance->initializeController(instance);
144 }
145
146 mutex_unlock(&mutex);
147 return instance;
148}
